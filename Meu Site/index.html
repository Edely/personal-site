from bs4 import BeautifulSoup
import mechanize

br = mechanize.Browser()
br.open("https://pt.wikipedia.org/wiki/Unidades_federativas_do_Brasil")
pagina_wikipedia = br.response().read()

soup = BeautifulSoup(pagina_wikipedia, "html5lib")
tabela_estados = soup.find_all("table", class_=["wikitable", "sortable", "jquery-tablesorter"])	

trs = tabela_estados[0].find_all("tr")
#i = 0
for tr in trs:
	tds = tr.find_all("td")
	
	try:
		estado = tds[1]
		populacao = tds[4]
		capital = tds[3]
		populacao = populacao.get_text()
		capital = capital.get_text()
		estado = estado.get_text()
		
		print estado
		print 'Capital:'
		print capital
		print 'População:'
		print populacao
		print '\n'
	except:
		pass
